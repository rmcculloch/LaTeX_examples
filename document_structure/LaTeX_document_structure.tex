%This .tex document shows how to create a basic document structure including the
%title page, table of contents, abstract and sections.
\documentclass[12pt]{article}

%Sets the table of contents depth.
\setcounter{tocdepth}{5}

%This command sets the depth to which numbers will appear for the sections. An
%argument of -1 means that no numbers will be displayed.
%\setcounter{secnumdepth}{-1}
\setcounter{secnumdepth}{5}

\begin{document}

\title{My Practice \LaTeX{} Document}
\author{Richard McCulloch} %You can add other author/s by adding \and authorname
%within the parentheses.
%Note that \maketitle will insert today's date by default anyway. You can leave
%date out by including \date{} with nothing in the parentheses.
\date{\today}
\maketitle

\newpage

%Needs to be compiled twice before it fully renders.
\tableofcontents

\newpage
